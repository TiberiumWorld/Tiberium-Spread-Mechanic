country_event = {   # tiberium_grow.0 // check growth
    id = tiberium_grow.0
    title = tiberium_grow.0.t
    desc = tiberium_grow.0.desc
    #picture = GFX_news_event_tiberium

    is_triggered_only = yes
    hidden = yes
    fire_only_once = no

    immediate = {
        if = {
            limit = { has_game_rule = { rule = tiberium_rule_growth option = LOW } }
            random_list = {
                5 = { country_event = {id = tiberium_grow.0 days = 1 } }
                95 = { }
            }
        }
        else_if = {
            limit = { has_game_rule = { rule = tiberium_rule_growth option = NORMAL } }
            random_list = {
                10 = { country_event = {id = tiberium_grow.0 days = 1 } }
                90 = { }
            }
        }
        else_if = {
            limit = { has_game_rule = { rule = tiberium_rule_growth option = HIGH } }
            random_list = {
                25 = { country_event = {id = tiberium_grow.0 days = 1 } }
                75 = { }
            }
        }
        else_if = {
            limit = { has_game_rule = { rule = tiberium_rule_growth option = EXTREAM } }
            random_list = {
                40 = { country_event = {id = tiberium_grow.0 days = 1 } }
                60 = { }
            }
        }
    }
    
    option = {
        name = tiberium_grow.0.a
    }
}
country_event = {   # tiberium_grow.1 // growth
    id = tiberium_grow.1
    title = tiberium_grow.1.t
    desc = tiberium_grow.1.desc
    #picture = GFX_news_event_tiberium

    is_triggered_only = yes
    hidden = yes
    fire_only_once = no

    immediate = {
        every_owned_state = {
            limit = {
                has_tiberium_infestation = yes
            }
            THIS = {
                increase_tiberium_infestation_level = yes
            }
        }
        # Damage structures
        every_owned_state = {
            limit = {
                OR = {
                    has_state_flag = tiberium_infestation_1_flag
                    has_state_flag = tiberium_infestation_2_flag
                    has_state_flag = tiberium_infestation_3_flag
                }
                NOT = { has_state_flag = tiberium_infestation_0_flag }
            }
            THIS = {
                damage_building = {
                    type = infrastructure
                    damage = 0.20
                }
                damage_building = {
                    type = arms_factory
                    damage = 0.20
                }
                damage_building = {
                    type = industrial_complex
                    damage = 0.20
                }
                damage_building = {
                    type = dockyard

                }
